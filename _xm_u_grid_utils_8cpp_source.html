52 void iWriteUGridToAsciiFile(BSHP<XmUGrid> a_ugrid, std::ostream& a_outStream)
53 {
54  a_outStream << "ASCII XmUGrid Version 1.0\n";
55 
56  // number of points
57  const VecPt3d& points = a_ugrid->GetLocations();
58  a_outStream << "NUM_POINTS " << points.size() << "\n";
59 
60  // points
61  for (size_t i = 0; i < points.size(); ++i)
62  {
63  const Pt3d& p = points[i];
64  a_outStream << " POINT " << STRstd(p.x) << " " << STRstd(p.y) << " " << STRstd(p.z) << "\n";
65  }
66 
67  // number of cell stream items
68  const VecInt& cellstream = a_ugrid->GetCellstream();
69  int cellstreamSize = (int)cellstream.size();
70  a_outStream << "NUM_CELL_ITEMS " << cellstreamSize << "\n";
71 
72  // cells
73  int currIdx = 0;
74  while (currIdx < cellstreamSize)
75  {
76  int cellType = cellstream[currIdx++];
77  int numItems = cellstream[currIdx++];
78  a_outStream << " CELL " << cellType;
79  if (cellType == -1)
80  {
81  currIdx += numItems;
82  }
83  else if (cellType == XMU_POLYHEDRON)
84  {
85  int numFaces = numItems;
86  a_outStream << " " << numFaces;
87  for (int faceIdx = 0; faceIdx < numFaces; ++faceIdx)
88  {
89  numItems = cellstream[currIdx++];
90  a_outStream << "\n " << numItems;
91  for (int itemIdx = 0; itemIdx < numItems; ++itemIdx)
92  {
93  a_outStream << " " << cellstream[currIdx++];
94  }
95  }
96  }
97  else
98  {
99  a_outStream << " " << numItems;
100  for (int itemIdx = 0; itemIdx < numItems; ++itemIdx)
101  {
102  a_outStream << " " << cellstream[currIdx++];
103  }
104  }
105  a_outStream << "\n";
106  }
107 } // iWriteUGridToAsciiFile

113 BSHP<XmUGrid> iReadUGridFromAsciiFile(std::istream& a_inStream)
114 {
115  if (a_inStream.eof())
116  {
117  return nullptr;
118  }
119 
120  std::string line;
121  std::getline(a_inStream, line);
122  if (line != "ASCII XmUGrid Version 1.0")
123  {
124  XM_LOG(xmlog::error, "Unsupported file version or file type: " + line);
125  return nullptr;
126  }
127 
128  std::string cardName;
129  VecPt3d points;
130  VecInt cells;
131  while (!a_inStream.eof())
132  {
133  cardName = "";
134  a_inStream >> cardName;
135  if (cardName == "NUM_POINTS")
136  {
137  int numPoints;
138  a_inStream >> numPoints;
139  points.reserve(numPoints);
140  }
141  else if (cardName == "POINT")
142  {
143  Pt3d pt;
144  a_inStream >> pt.x >> pt.y >> pt.z;
145  points.push_back(pt);
146  }
147  else if (cardName == "NUM_CELL_ITEMS")
148  {
149  int numCellItems;
150  a_inStream >> numCellItems;
151  cells.reserve(numCellItems);
152  }
153  else if (cardName == "CELL")
154  {
155  int cellType;
156  a_inStream >> cellType;
157  cells.push_back(cellType);
158  if (cellType == XMU_POLYHEDRON)
159  {
160  int numFaces;
161  a_inStream >> numFaces;
162  cells.push_back(numFaces);
163  for (int faceIdx = 0; faceIdx < numFaces; ++faceIdx)
164  {
165  int numPoints;
166  a_inStream >> numPoints;
167  cells.push_back(numPoints);
168  for (int i = 0; i < numPoints; ++i)
169  {
170  int ptIdx;
171  a_inStream >> ptIdx;
172  cells.push_back(ptIdx);
173  }
174  }
175  }
176  else
177  {
178  int numPoints;
179  a_inStream >> numPoints;
180  cells.push_back(numPoints);
181  for (int i = 0; i < numPoints; ++i)
182  {
183  int ptIdx;
184  a_inStream >> ptIdx;
185  cells.push_back(ptIdx);
186  }
187  }
188  }
189  }
190 
191  BSHP<XmUGrid> ugrid = XmUGrid::New(points, cells);
192  return ugrid;
193 } // iReadUGridFromAsciiFile
