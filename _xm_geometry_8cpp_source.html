70 VecPt3d ConvexHull(const std::vector<Pt3<double>>& a_points)
71 {
72  VecPt3d points = a_points;
73  size_t n = points.size(), j = 0;
74  if (n <= 3)
75  return points;
76  VecPt3d convexHull(2 * n);
77 
78  // Sort points lexicographically
79  sort(points.begin(), points.end());
80 
81  // Build lower hull
82  for (size_t i = 0; i < n; ++i)
83  {
84  while (j >= 2 && iCross(convexHull[j - 2], convexHull[j - 1], points[i]) <= 0)
85  j--;
86  convexHull[j++] = points[i];
87  }
88 
89  // Build upper hull
90  for (size_t i = n - 1, t = j + 1; i > 0; --i)
91  {
92  while (j >= t && iCross(convexHull[j - 2], convexHull[j - 1], points[i - 1]) <= 0)
93  j--;
94  convexHull[j++] = points[i - 1];
95  }
96 
97  convexHull.resize(j - 1);
98  return convexHull;
99 } // ConvexHull

151 bool DoLineSegmentsCross(const Pt3<double>& a_segment1Point1,
152  const Pt3<double>& a_segment1Point2,
153  const Pt3<double>& a_segment2Point1,
154  const Pt3<double>& a_segment2Point2)
155 {
156  // Boundary case checks
157  // Any of the points from line segment 1 are the same as any points from line segment 2
158  if ((a_segment1Point1 == a_segment2Point1 || a_segment1Point1 == a_segment2Point2) &&
159  (a_segment1Point2 == a_segment2Point1 || a_segment1Point2 == a_segment2Point2))
160  return true;
161 
162  // The segments AB and CD intersect if and only if both of the following are true:
163  //
164  // A and B lie on different sides of the line through C and D
165  // C and D lie on different sides of the line through A and B
166  // These two conditions can be tested for using the notion of a scalar cross product(formulas
167  // below).
168 
169  // is true if and only if the scalar cross products CA->�CD-> and CB->�CD-> have opposite signs.
170  // is true if and only if the scalar cross products AC->�AB-> and AD->�AB-> have opposite signs.
171 
172  // Conclusion: the line segments intersect if and only if both are negative
173  double result1 = iCross(a_segment2Point1, a_segment1Point1, a_segment2Point2);
174  double result2 = iCross(a_segment2Point1, a_segment1Point2, a_segment2Point2);
175  double result3 = iCross(a_segment1Point1, a_segment2Point1, a_segment1Point2);
176  double result4 = iCross(a_segment2Point1, a_segment2Point2, a_segment1Point2);
177 
178  return (result1 * result2 < 0 && result3 * result4 < 0);
