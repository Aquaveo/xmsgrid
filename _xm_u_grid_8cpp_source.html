598 //------------------------------------------------------------------------------
606 //------------------------------------------------------------------------------
607 bool gmEqualPointsXY(double x1, double y1, double x2, double y2, double tolerance)
608 {
609  double dx = fabs(x1 - x2);
610  double dy = fabs(y1 - y2);
611  if (dx > tolerance || dy > tolerance)
612  return false;
613  else if (sqrt(dx * dx + dy * dy) <= tolerance)
614  return true;
615  else
616  return false;
617 } // gmEqualPointsXY
618 //------------------------------------------------------------------------------
633 //------------------------------------------------------------------------------
634 bool gmOnLineWithTol(const Pt3d& p1,
635  const Pt3d& p2,
636  const double x,
637  const double y,
638  const double tol)
639 {
640  // compute vector components
641  double dx = p2.x - p1.x;
642  double dy = p2.y - p1.y;
643  double mag = sqrt((dx * dx) + (dy * dy));
644  // check for extremely small segment
645  if (mag <= tol)
646  return gmEqualPointsXY(p1.x, p1.y, x, y, 1.0e-7);
647  else
648  {
649  double a = -dy / mag;
650  double b = dx / mag;
651  double c = -a * p2.x - b * p2.y;
652  // compute distance from line to (x,y)
653  double d = a * x + b * y + c;
654  return fabs(d) <= tol;
655  }
656 } // gmOnLineWithTol
657 //------------------------------------------------------------------------------
664 //------------------------------------------------------------------------------
665 double gmPolygonArea(const Pt3d* pts, size_t npoints)
666 {
667  size_t id;
668  double area = 0.0;
669 
670  if (npoints < 3)
671  return area;
672 
673  /* original method with precision errors
674  for (id = 0; id < npoints; id++)
675  {
676  if (id != (npoints - 1))
677  {
678  area += (pts[id].x * pts[id + 1].y);
679  area -= (pts[id].y * pts[id + 1].x);
680  }
681  else
682  {
683  area += (pts[id].x * pts[0].y);
684  area -= (pts[id].y * pts[0].x);
685  }
686  }
687  area /= 2.0;
688  */
689 
690  // AKZ 2/15/2018
691  // I changed the implementation to translate the polygon
692  // so that the first point is at the origin
693  // Reduces round off error due to large coordinates
694  // Reduces the number of computations because the first and last
695  // computations in the loop would be 0.0
696  VecDbl x, y;
697  double x0 = pts[0].x;
698  double y0 = pts[0].y;
699  for (id = 1; id < npoints; id++)
700  {
701  x.push_back((pts[id].x - x0));
702  y.push_back((pts[id].y - y0));
703  }
704  for (id = 0; id < npoints - 2; id++)
705  {
706  area += (x[id] * y[id + 1]);
707  area -= (y[id] * x[id + 1]);
708  }
709  area /= 2.0;
710 
711  return (area);
712 } // gmPolygonArea
713 //------------------------------------------------------------------------------
717 //------------------------------------------------------------------------------
718 Pt3d gmComputePolygonCentroid(const VecPt3d& pts)
719 {
720  Pt3d centroid;
721  if (pts.empty())
722  return centroid;
723  // get offset to use in calculation below to fix precision issues
724  double xMax = XM_DBL_LOWEST, yMax = XM_DBL_LOWEST, xMin = XM_DBL_HIGHEST, yMin = XM_DBL_HIGHEST;
725  size_t i = 0;
726  for (i = 0; i < pts.size(); ++i)
727  {
728  double x = pts[i].x;
729  double y = pts[i].y;
730  xMax = (x > xMax) ? x : xMax;
731  yMax = (y > yMax) ? y : yMax;
732  xMin = (x < xMin) ? x : xMin;
733  yMin = (y < yMin) ? y : yMin;
734  }
735  double xOffset = (xMax + xMin) / 2.0;
736  double yOffset = (yMax + yMin) / 2.0;
737  // For all vertices except last
738  double signedArea = 0.0;
739  for (i = 0; i < pts.size() - 1; ++i)
740  {
741  double x0 = pts[i].x - xOffset;
742  double y0 = pts[i].y - yOffset;
743  double x1 = pts[i + 1].x - xOffset;
744  double y1 = pts[i + 1].y - yOffset;
745  double a = x0 * y1 - x1 * y0;
746  signedArea += a;
747  centroid.x += (x0 + x1) * a;
748  centroid.y += (y0 + y1) * a;
749  }
750  // Do last vertex
751  double x0 = pts[i].x - xOffset;
752  double y0 = pts[i].y - yOffset;
753  double x1 = pts[0].x - xOffset;
754  double y1 = pts[0].y - yOffset;
755  double a = x0 * y1 - x1 * y0;
756  signedArea += a;
757  centroid.x += (x0 + x1) * a;
758  centroid.y += (y0 + y1) * a;
759  signedArea *= 0.5;
760  centroid.x /= (6.0 * signedArea);
761  centroid.y /= (6.0 * signedArea);
762  centroid.x += xOffset;
763  centroid.y += yOffset;
764  return centroid;
765 } // gmComputePolygonCentroid
766 //------------------------------------------------------------------------------
768 //------------------------------------------------------------------------------
769 void iMergeSegmentsToPoly(const VecPt3d& segments, VecPt3d& polygon)
770 {
771  if (segments.empty())
772  return;
773 
774  VecPt3d points = iGetUniquePoints(segments);
775  std::vector<std::pair<Pt3d, Pt3d>> segs = iGetUniqueSegments(segments);
776 
777  // if any point is on any other segment split up the segment
778  for (auto pIt = points.begin(); pIt != points.end(); ++pIt)
779  {
780  size_t segsSize = segs.size();
781  for (size_t i = 0; i < segsSize; ++i)
782  {
783  Pt3d pt1 = segs[i].first;
784  Pt3d pt2 = segs[i].second;
785  if (*pIt != pt1 && *pIt != pt2 && iPointInSegmentBounds(*pIt, pt1, pt2))
786  {
787  Pt3d diff = pt1 - pt2;
788  double distance = std::max(fabs(diff.x), fabs(diff.y));
789  double tolerance = distance / 1.0e6;
790  if (gmOnLineWithTol(pt1, pt2, pIt->x, pIt->y, tolerance))
791  {
792  segs.push_back(std::pair<Pt3d, Pt3d>(*pIt, pt2));
793  segs[i].second = *pIt;
794  }
795  }
796  }
797  }
798  std::sort(segs.begin(), segs.end());
799  auto segIt = std::unique(segs.begin(), segs.end());
800  segs.resize(segIt - segs.begin());
801 
802  iBuildPolygon(segs, polygon);
803 
804  if (polygon.size() > 3 && polygon.front() == polygon.back())
805  {
806  polygon.pop_back();
807  double area = gmPolygonArea(&polygon[0], (int)polygon.size());
808  if (area < 0)
809  std::reverse(polygon.begin(), polygon.end());
810  }
811  else
812  {
813  XM_ASSERT(0);
814  polygon.clear();
815  }
816 } // iMergeSegmentsToPoly
