594  }
595  std::sort(segs.begin(), segs.end());
596  auto segIt = std::unique(segs.begin(), segs.end());
597  segs.resize(segIt - segs.begin());

600 //------------------------------------------------------------------------------
608 //------------------------------------------------------------------------------
609 bool gmEqualPointsXY(double x1, double y1, double x2, double y2, double tolerance)
610 {
611  double dx = fabs(x1 - x2);
612  double dy = fabs(y1 - y2);
613  if (dx > tolerance || dy > tolerance)
614  return false;
615  else if (sqrt(dx * dx + dy * dy) <= tolerance)
616  return true;
617  else
618  return false;
619 } // gmEqualPointsXY
620 //------------------------------------------------------------------------------
635 //------------------------------------------------------------------------------
636 bool gmOnLineWithTol(const Pt3d& p1,
637  const Pt3d& p2,
638  const double x,
639  const double y,
640  const double tol)
641 {
642  // compute vector components
643  double dx = p2.x - p1.x;
644  double dy = p2.y - p1.y;
645  double mag = sqrt((dx * dx) + (dy * dy));
646  // check for extremely small segment
647  if (mag <= tol)
648  return gmEqualPointsXY(p1.x, p1.y, x, y, 1.0e-7);
649  else
650  {
651  double a = -dy / mag;
652  double b = dx / mag;
653  double c = -a * p2.x - b * p2.y;
654  // compute distance from line to (x,y)
655  double d = a * x + b * y + c;
656  return fabs(d) <= tol;
657  }
658 } // gmOnLineWithTol
659 //------------------------------------------------------------------------------
666 //------------------------------------------------------------------------------
667 double gmPolygonArea(const Pt3d* pts, size_t npoints)
668 {
669  size_t id;
670  double area = 0.0;
671 
672  if (npoints < 3)
673  return area;
674 
675  /* original method with precision errors
676  for (id = 0; id < npoints; id++)
677  {
678  if (id != (npoints - 1))
679  {
680  area += (pts[id].x * pts[id + 1].y);
681  area -= (pts[id].y * pts[id + 1].x);
682  }
683  else
684  {
685  area += (pts[id].x * pts[0].y);
686  area -= (pts[id].y * pts[0].x);
687  }
688  }
689  area /= 2.0;
690  */
691 
692  // AKZ 2/15/2018
693  // I changed the implementation to translate the polygon
694  // so that the first point is at the origin
695  // Reduces round off error due to large coordinates
696  // Reduces the number of computations because the first and last
697  // computations in the loop would be 0.0
698  VecDbl x, y;
699  double x0 = pts[0].x;
700  double y0 = pts[0].y;
701  for (id = 1; id < npoints; id++)
702  {
703  x.push_back((pts[id].x - x0));
704  y.push_back((pts[id].y - y0));
705  }
706  for (id = 0; id < npoints - 2; id++)
707  {
708  area += (x[id] * y[id + 1]);
709  area -= (y[id] * x[id + 1]);
710  }
711  area /= 2.0;
712 
713  return (area);
714 } // gmPolygonArea
715 //------------------------------------------------------------------------------
719 //------------------------------------------------------------------------------
720 Pt3d gmComputePolygonCentroid(const VecPt3d& pts)
721 {
722  Pt3d centroid;
723  if (pts.empty())
724  return centroid;
725  // get offset to use in calculation below to fix precision issues
726  double xMax = XM_DBL_LOWEST, yMax = XM_DBL_LOWEST, xMin = XM_DBL_HIGHEST, yMin = XM_DBL_HIGHEST;
727  size_t i = 0;
728  for (i = 0; i < pts.size(); ++i)
729  {
730  double x = pts[i].x;
731  double y = pts[i].y;
732  xMax = (x > xMax) ? x : xMax;
733  yMax = (y > yMax) ? y : yMax;
734  xMin = (x < xMin) ? x : xMin;
735  yMin = (y < yMin) ? y : yMin;
736  }
737  double xOffset = (xMax + xMin) / 2.0;
738  double yOffset = (yMax + yMin) / 2.0;
739  // For all vertices except last
740  double signedArea = 0.0;
741  for (i = 0; i < pts.size() - 1; ++i)
742  {
743  double x0 = pts[i].x - xOffset;
744  double y0 = pts[i].y - yOffset;
745  double x1 = pts[i + 1].x - xOffset;
746  double y1 = pts[i + 1].y - yOffset;
747  double a = x0 * y1 - x1 * y0;
748  signedArea += a;
749  centroid.x += (x0 + x1) * a;
750  centroid.y += (y0 + y1) * a;
751  }
752  // Do last vertex
753  double x0 = pts[i].x - xOffset;
754  double y0 = pts[i].y - yOffset;
755  double x1 = pts[0].x - xOffset;
756  double y1 = pts[0].y - yOffset;
757  double a = x0 * y1 - x1 * y0;
758  signedArea += a;
759  centroid.x += (x0 + x1) * a;
760  centroid.y += (y0 + y1) * a;
761  signedArea *= 0.5;
762  centroid.x /= (6.0 * signedArea);
763  centroid.y /= (6.0 * signedArea);
764  centroid.x += xOffset;
765  centroid.y += yOffset;
766  return centroid;
767 } // gmComputePolygonCentroid

770 //------------------------------------------------------------------------------
771 void iMergeSegmentsToPoly(const VecPt3d& segments, VecPt3d& polygon)
772 {
773  if (segments.empty())
774  return;
775 
776  VecPt3d points = iGetUniquePoints(segments);
777  std::vector<std::pair<Pt3d, Pt3d>> segs = iGetUniqueSegments(segments);
778 
779  // if any point is on any other segment split up the segment
780  for (auto pIt = points.begin(); pIt != points.end(); ++pIt)
781  {
782  size_t segsSize = segs.size();
783  for (size_t i = 0; i < segsSize; ++i)
784  {
785  Pt3d pt1 = segs[i].first;
786  Pt3d pt2 = segs[i].second;
787  if (*pIt != pt1 && *pIt != pt2 && iPointInSegmentBounds(*pIt, pt1, pt2))
788  {
789  Pt3d diff = pt1 - pt2;
790  double distance = std::max(fabs(diff.x), fabs(diff.y));
791  double tolerance = distance / 1.0e6;
792  if (gmOnLineWithTol(pt1, pt2, pIt->x, pIt->y, tolerance))
793  {
794  segs.push_back(std::pair<Pt3d, Pt3d>(*pIt, pt2));
795  segs[i].second = *pIt;
796  }
797  }
798  }
799  }
800  std::sort(segs.begin(), segs.end());
801  auto segIt = std::unique(segs.begin(), segs.end());
802  segs.resize(segIt - segs.begin());
803 
804  iBuildPolygon(segs, polygon);
805 
806  if (polygon.size() > 3 && polygon.front() == polygon.back())
807  {
808  polygon.pop_back();
809  double area = gmPolygonArea(&polygon[0], (int)polygon.size());
810  if (area < 0)
811  std::reverse(polygon.begin(), polygon.end());
812  }
813  else
814  {
815  XM_ASSERT(0);
816  polygon.clear();
817  }
818 } // iMergeSegmentsToPoly
