591  for (size_t j = 1; j != segs.size(); ++j)

599  }
600  else if (segs[j].second == toMatch)
601  {
602  polygon.push_back(segs[j].first);
603  placed[j] = true;

611 //------------------------------------------------------------------------------
612 template <typename T>
613 std::vector<std::pair<T, T>> iGetUniqueSegments(const std::vector<T>& segments)
614 {
615  std::vector<std::pair<T, T>> segs;
616  for (size_t i = 0; i < segments.size(); i += 2)
617  {
618  T a = segments[i];
619  T b = segments[i + 1];
620  if (a < b)
621  segs.push_back(std::pair<T, T>(a, b));
622  else
623  segs.push_back(std::pair<T, T>(b, a));
624  }
625  std::sort(segs.begin(), segs.end());
626  auto segIt = std::unique(segs.begin(), segs.end());
627  segs.resize(segIt - segs.begin());
628  return segs;
629 } // iGetUniqueSegments
630 //------------------------------------------------------------------------------
638 //------------------------------------------------------------------------------
639 bool gmEqualPointsXY(double x1, double y1, double x2, double y2, double tolerance)
640 {
641  double dx = fabs(x1 - x2);
642  double dy = fabs(y1 - y2);
643  if (dx > tolerance || dy > tolerance)
644  return false;
645  else if (sqrt(dx * dx + dy * dy) <= tolerance)
646  return true;
647  else
648  return false;
649 } // gmEqualPointsXY
650 //------------------------------------------------------------------------------
665 //------------------------------------------------------------------------------
666 bool gmOnLineWithTol(const Pt3d& p1,
667  const Pt3d& p2,
668  const double x,
669  const double y,
670  const double tol)
671 {
672  // compute vector components
673  double dx = p2.x - p1.x;
674  double dy = p2.y - p1.y;
675  double mag = sqrt((dx*dx) + (dy*dy));
676  // check for extremely small segment
677  if (mag <= tol)
678  return gmEqualPointsXY(p1.x, p1.y, x, y, 1.0e-7);
679  else
680  {
681  double a = -dy / mag;
682  double b = dx / mag;
683  double c = -a * p2.x - b * p2.y;
684  // compute distance from line to (x,y)
685  double d = a * x + b * y + c;
686  return fabs(d) <= tol;
687  }
688 } // gmOnLineWithTol
689 //------------------------------------------------------------------------------
696 //------------------------------------------------------------------------------
697 double gmPolygonArea(const Pt3d* pts, size_t npoints)
698 {
699  size_t id;
700  double area = 0.0;
701 
702  if (npoints < 3)
703  return area;
704 
705  /* original method with precision errors
706  for (id = 0; id < npoints; id++)
707  {
708  if (id != (npoints - 1))
709  {
710  area += (pts[id].x * pts[id + 1].y);
711  area -= (pts[id].y * pts[id + 1].x);
712  }
713  else
714  {
715  area += (pts[id].x * pts[0].y);
716  area -= (pts[id].y * pts[0].x);
717  }
718  }
719  area /= 2.0;
720  */
721 
722  // AKZ 2/15/2018
723  // I changed the implementation to translate the polygon
724  // so that the first point is at the origin
725  // Reduces round off error due to large coordinates
726  // Reduces the number of computations because the first and last
727  // computations in the loop would be 0.0
728  VecDbl x, y;
729  double x0 = pts[0].x;
730  double y0 = pts[0].y;
731  for (id = 1; id < npoints; id++)
732  {
733  x.push_back((pts[id].x - x0));
734  y.push_back((pts[id].y - y0));
735  }
736  for (id = 0; id < npoints - 2; id++)
737  {
738  area += (x[id] * y[id + 1]);
739  area -= (y[id] * x[id + 1]);
740  }
741  area /= 2.0;
742 
743  return (area);
744 } // gmPolygonArea
745 //------------------------------------------------------------------------------
749 //------------------------------------------------------------------------------
750 Pt3d gmComputePolygonCentroid(const VecPt3d& pts)
751 {
752  Pt3d centroid;
753  if (pts.empty())
754  return centroid;
755  // get offset to use in calculation below to fix precision issues
756  double xMax = XM_DBL_LOWEST, yMax = XM_DBL_LOWEST, xMin = XM_DBL_HIGHEST, yMin = XM_DBL_HIGHEST;
757  size_t i = 0;
758  for (i = 0; i < pts.size(); ++i)
759  {
760  double x = pts[i].x;
761  double y = pts[i].y;
762  xMax = (x > xMax) ? x : xMax;
763  yMax = (y > yMax) ? y : yMax;
764  xMin = (x < xMin) ? x : xMin;
765  yMin = (y < yMin) ? y : yMin;
766  }
767  double xOffset = (xMax + xMin) / 2.0;
768  double yOffset = (yMax + yMin) / 2.0;
769  // For all vertices except last
770  double signedArea = 0.0;
771  for (i = 0; i < pts.size() - 1; ++i)
772  {
773  double x0 = pts[i].x - xOffset;
774  double y0 = pts[i].y - yOffset;
775  double x1 = pts[i + 1].x - xOffset;
776  double y1 = pts[i + 1].y - yOffset;
777  double a = x0 * y1 - x1 * y0;
778  signedArea += a;
779  centroid.x += (x0 + x1) * a;
780  centroid.y += (y0 + y1) * a;
781  }
782  // Do last vertex
783  double x0 = pts[i].x - xOffset;
784  double y0 = pts[i].y - yOffset;
785  double x1 = pts[0].x - xOffset;
786  double y1 = pts[0].y - yOffset;
787  double a = x0 * y1 - x1 * y0;
788  signedArea += a;
789  centroid.x += (x0 + x1) * a;
790  centroid.y += (y0 + y1) * a;
791  signedArea *= 0.5;
792  centroid.x /= (6.0 * signedArea);
793  centroid.y /= (6.0 * signedArea);
794  centroid.x += xOffset;
795  centroid.y += yOffset;
796  return centroid;
797 } // gmComputePolygonCentroid

800 //------------------------------------------------------------------------------
801 void iMergeSegmentsToPoly(const VecPt3d& segments, VecPt3d& polygon)
802 {
803  if (segments.empty())
804  return;
805 
806  VecPt3d points = iGetUniquePoints(segments);
807  std::vector<std::pair<Pt3d, Pt3d>> segs = iGetUniqueSegments(segments);
808 
809  // if any point is on any other segment split up the segment
810  for (auto pIt = points.begin(); pIt != points.end(); ++pIt)
811  {
812  size_t segsSize = segs.size();
813  for (size_t i = 0; i < segsSize; ++i)
814  {
815  Pt3d pt1 = segs[i].first;
816  Pt3d pt2 = segs[i].second;
817  if (*pIt != pt1 && *pIt != pt2 && iPointInSegmentBounds(*pIt, pt1, pt2))
818  {
819  Pt3d diff = pt1 - pt2;
820  double distance = std::max(fabs(diff.x), fabs(diff.y));
821  double tolerance = distance / 1.0e6;
822  if (gmOnLineWithTol(pt1, pt2, pIt->x, pIt->y, tolerance))
823  {
824  segs.push_back(std::pair<Pt3d, Pt3d>(*pIt, pt2));
825  segs[i].second = *pIt;
826  }
827  }
828  }
829  }
830  std::sort(segs.begin(), segs.end());
831  auto segIt = std::unique(segs.begin(), segs.end());
832  segs.resize(segIt - segs.begin());
833 
834  iBuildPolygon(segs, polygon);
835 
836  if (polygon.size() > 3 && polygon.front() == polygon.back())
837  {
838  polygon.pop_back();
839  double area = gmPolygonArea(&polygon[0], (int)polygon.size());
840  if (area < 0)
841  std::reverse(polygon.begin(), polygon.end());
842  }
843  else
844  {
845  XM_ASSERT(0);
846  polygon.clear();
847  }
848 } // iMergeSegmentsToPoly
