57  def get_hexahedron_ugrid(rows, cols, lays, origin):
58  points = [None] * rows * cols * lays
59  curr_id = 0
60  for k in range(0, lays):
61  for i in range(0, rows):
62  for j in range(0, cols):
63  points[curr_id] = (j + origin[0], rows - i - 1 + origin[1], lays - k - 1 + origin[2])
64  curr_id += 1
65  cells = [None] * (rows - 1) * (cols - 1) * (lays - 1) * 10
66  num_in_layer = rows * cols
67  curr_id = 0
68  for k in range(0, lays - 1):
69  lay_offset = num_in_layer * k
70  for i in range(0, rows - 1):
71  for j in range(0, cols - 1):
72  cells[curr_id] = XmUGrid.xmugrid_celltype_enum.XMU_HEXAHEDRON
73  cells[curr_id + 1] = 8
74  cells[curr_id + 2] = j + cols * i + lay_offset
75  cells[curr_id + 3] = j + cols * (i + 1) + lay_offset
76  cells[curr_id + 4] = j + 1 + cols * (i + 1) + lay_offset
77  cells[curr_id + 5] = j + 1 + cols * i + lay_offset
78  cells[curr_id + 6] = j + cols * i + lay_offset + num_in_layer
79  cells[curr_id + 7] = j + cols * (i + 1) + lay_offset + num_in_layer
80  cells[curr_id + 8] = j + 1 + cols * (i + 1) + lay_offset + num_in_layer
81  cells[curr_id + 9] = j + 1 + cols * i + lay_offset + num_in_layer
82  curr_id += 10
83  xu = XmUGrid(points, cells)
