54 std::vector<Pt3d> ConvexHull(std::vector<Pt3d> a_points)
55 {
56  size_t n = a_points.size(), j = 0;
57  if (n <= 3)
58  return a_points;
59  std::vector<Pt3d> convexHull(2 * n);
60 
61  // Sort points lexicographically
62  sort(a_points.begin(), a_points.end());
63 
64  // Build lower hull
65  for (size_t i = 0; i < n; ++i)
66  {
67  while (j >= 2 && cross(convexHull[j - 2], convexHull[j - 1], a_points[i]) <= 0)
68  j--;
69  convexHull[j++] = a_points[i];
70  }
71 
72  // Build upper hull
73  for (size_t i = n - 1, t = j + 1; i > 0; --i)
74  {
75  while (j >= t && cross(convexHull[j - 2], convexHull[j - 1], a_points[i - 1]) <= 0)
76  j--;
77  convexHull[j++] = a_points[i - 1];
78  }
79 
80  convexHull.resize(j - 1);
